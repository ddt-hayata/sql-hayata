/* 
 * 店舗テーブル（shop）と月間売上テーブル（sales）から2012年12月の売上高が
 * 店舗売上平均（2012年12月の平均）を下回っている店舗名とその売上高をあげてみましょう。
 */

 // 問題
 
 // 回答
 SELECT sh.s_name, sa.s_value
 FROM shop AS sh
 INNNER JOIN sales AS sa
 ON sh.s_id = sa.s_id
 WHERE sa.s_data = '2012-12'
 AND sa.s_value < 
     (
         SELECT AVG(s_value)
         FROM sales
         WHERE s_data = '2012-12'
     )

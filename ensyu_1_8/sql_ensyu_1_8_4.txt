/* 
 * 以下は、アクセス記録テーブル（access_log）からアクセス日時の新しい順に
 * 10件のログデータを取得するためのSQL命令です（取得列はpage_id, referer, ip_address）が、
 * 2点誤りがあります。誤りを指摘してください。
 */

 // 問題
 SELECT page_id, referer, ip_address
 FROM access_log
 ORDER BY access_date ASC LIMIT 10, 0
 ;

 // 回答
 SELECT page_id, referer, ip_address
 FROM access_log
 ORDER BY access_date DESC LIMIT 0, 10　※「access_date ASC」→「access_date DESC」、「LIMIT 10, 0」→「LIMIT 0, 10」へ修正
 ;

 // 問題
 DELETE FROM
 employee
 WHERE
 ➀
 (
     SELECT
     depart.depart_id
     FROM
     ➁
     WHERE
     ➂
 )
 ;

 // 回答
 DELETE FROM
 employee
 WHERE
 NOT EXITS
 (
     SELECT
     depart.depart_id
     FROM
     depart
     WHERE
     depart.depart_id = employee.depart_id
 )
 ;

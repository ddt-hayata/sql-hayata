 // 問題
 ➀
 employee
 SET
 depart_id = 'Z99'
 WHERE
 ➁
 （
 SELECT
 *
 FROM
 depart
 WHERE
 ➂
 ）
 ;

 // 回答
 UPDATE
 employee
 SET
 depart_id = 'Z99'
 WHERE
 NOT EXITS
 （
 SELECT
 *
 FROM
 depart
 WHERE
 employee.depart_id = depart.depart_id
 ）
 ;

/* 
 * 貸出記録テーブル（rental）をISBNコードでグループ化し、それぞれの「貸出数」列を出力します。
 * その際、カウント数が10以上ならば「好評」、5以上10未満ならば「普通」、5未満ならば「不評」とする
 * 「評価」列も合わせて出力してみましょう。以下の空欄を埋めて、SQL命令を完成させてみましょう。
 */

 // 問題
 SELECT isbn, ➀
 CASE
 ➁ count(*) >= 10 ➂ '好評'
 ➁ count(*) >= 5 ➂ '普通'
 ELSE '不評'
 END AS 評価
 FROM rental
 ➃ ISBN
 ;
 
 // 回答
 SELECT isbn, count(*) AS 貸出数
 CASE
 WHEN count(*) >= 10 THEN '好評'
 WHEN count(*) >= 5 THEN '普通'
 ELSE '不評'
 END AS 評価
 FROM rental
 GROUP BY isbn
 ;

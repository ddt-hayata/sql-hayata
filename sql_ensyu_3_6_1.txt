/* 
 * 所属部署テーブル（depart）にない所属部署に属する社員を社員テーブル（employee）から抽出し、
 * 所属部署コードを「Z99（未登録）」い変更しましょう。以下の空欄を埋めて、SQL命令を完成させてください。
 */
 
 // 問題
 UPDATE
 employee
 SET
 ➀
 WHERE
 depart
 ➁
 （
 SELECT
 ➂
 FROM
 depart_id
 ）
 ;

 // 回答
 UPDATE
 employee
 SET
 depart_id = 'Z99'
 WHERE
 depart_id
 NOT IN
 （
 SELECT
 depart_id
 FROM
 depart
 ）
 ;

/* 
 * 貸し出し記録テーブル（rental）から未返却の書籍情報を貸出日について昇順で抽出してみましょう。
 * 取得列は、ISBNコードと貸出日（YY/MM/DDの形式）、別名はそれぞれ「ISBNコード」「貸出日」とします。
 * 以下の空欄を埋めて、SQL命令を完成させてください。
 */

 // 問題
 SELECT isbn AS ISBNコード, ➀
 FROM rental
 WHERE returned = 0 ➁ rental_date ASC
 ;

 // 回答
 SELECT isbn AS ISBNコード, DATE_FORMAT(rental_date, '%y年 %m月 %d日') AS 貸出日
 FROM rental
 WHERE returned = 0
 ORDER BY rental_date ASC
 ;

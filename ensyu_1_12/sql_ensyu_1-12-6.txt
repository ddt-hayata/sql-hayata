/* 
 * アクセス記録テーブル（access_log）からアクセス日付が2013/01/01以降のものについて
 * リンク元URLごとにあくせすすうを算出してみましょう。その際、アクセス数が5件未満のもののみをアクセス数降順で出力してください。
 * 以下の空欄を埋めて、SQL命令を完成させてください。
 */

// 問題
SELECT referer, ① FROM access_log WHERE ② GROUP BY referer ③ ① ＜ 5 ORDER BY ① ④;

// 回答
SELECT referer, COUNT(*) FROM access_log WHERE access_date >= '2013-01-01' GROUP BY referer HAVING COUNT(*) ＜ 5 ORDER BY COUNT(*) DESC;

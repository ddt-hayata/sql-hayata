/* 
 * アクセス記録テーブル（access_log）からアクセス日付が2013/01/01以降のものについて
 * リンク元URLごとにあくせすすうを算出してみましょう。その際、アクセス数が5件未満のもののみをアクセス数降順で出力してください。
 * 以下の空欄を埋めて、SQL命令を完成させてください。
 */

 // 問題
 SELECT referer, ➀
 FROM access_log
 WHERE ➁
 GROUP BY referer
 ➂ ➀ ＜ 5
 ORDER BY ➀ ➃
 ;

 // 回答
 SELECT referer, COUNT(*)
 FROM access_log
 WHERE access_date >= '2013-01-01'
 GROUP BY referer
 HAVING COUNT(*) ＜ 5
 ORDER BY COUNT(*) DESC
 ;

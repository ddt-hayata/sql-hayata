/* 
 * 以下は、社員テーブル（employee）から所属部署ごとの女性の人数を求め、3人以上の部署だけ
 * 出力するためのSQL命令ですが、2つ誤りがあります。誤りを指摘してください。
 */

 // 問題
 SELECT depart_id, COUNT(*)
 FROM employee
 WHERE sex = 2
 ORDER BY depart_id
 WHERE COUNT(*) >= 3
 ;

 // 回答
 SELECT depart_id, COUNT(*)
 FROM employee
 WHERE sex = 2
 GROUP BY depart_id　※「ORDER BY」→「GROUP BY」へ修正
 HAVING COUNT(*) >= 3　※「WHERE COUNT(*) >= 3」→「HAVING COUNT(*) >= 3　へ修正
 ;

/* 
 * 社員テーブル（employee）からそれぞれの社員氏名と上司氏名を、社員コードについて昇順に取り出してみましょう。
 * なお、取り出す列には別名として「社員氏名」「上司氏名」とし、上司がいない社員についても情報は取り出すことにします。
 */

 // 問題
 
 // 回答
 SELECT CONCAT(e1.l_name, el.f_name) AS 社員氏名, CONCAT(e2.l_name, e2.f_name) AS 上司氏名
 FROM employee AS e1
 LEFT JOIN employee AS e2
 ON e1.b_id = e2.s_id
 ORDER BY e1.s_id ASC

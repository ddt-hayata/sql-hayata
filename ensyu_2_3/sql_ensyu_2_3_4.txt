/* 
 * 以下は、書籍目次テーブル（contents）から現在のコンテンツコードと前のコンテンツコードを、
 * 現在のコンテンツコードについて昇順で取り出すためのSQL命令ですが、誤りが2点あります。誤りを指摘してください。
 */

 // 問題
 SELECT cp.c_id AS 前のページ, cn.c_id AS 現在のページ
 FROM contents AS cp
 OUTER JOIN contents AS cn
 ON cp.next_id <> cn.c_id
 OEDER BY cn.c_id ASC
 
 // 回答
 SELECT cp.c_id AS 前のページ, cn.c_id AS 現在のページ
 FROM contents AS cp
 INNNER JOIN contents AS cn　※「OUTER」→「INNNER」に修正
 ON cp.next_id = cn.c_id　※「<>」→「=」に修正
 OEDER BY cn.c_id ASC

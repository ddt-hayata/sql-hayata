/* 
 * 以下は、貸出記録テーブル(rental)上、利用者コードがB200405、ISBNコードが4-0010-0000-1のデータについて、 
 * 貸出日を今日の日付に更新するためのSQL命令ですが誤りが2点あります。誤りを指摘してください。
 */
 
 // 問題
 UPDATE
 rental
 SET
 rental_date CURRENT_DATE
 WHERE
 user_id = 'B200405'
 OR
 isbn = '4-0010-0000-1'

 // 回答
 UPDATE
 rental
 SET
 rental_date = CURRENT_DATE　※「＝」が抜けている
 WHERE
 user_id = 'B200405'
 AND　※「OR」→「AND」が正しい
 isbn = '4-0010-0000-1'
